Register User

METHOD: GET

ENDPOINT: /api/auth/register/

Request

{
    "username": "demo",
    "password": "password"
}

Response

{
"id": 1,
"username": "demo",
}

Login User

METHOD: POST

ENDPOINT: /api/auth/login/

Request

    {
        "username": "demo",
        "password": "password"
    }

Response

    {
        "id": 1,
        "username": "demo",
    }

Logout User

METHOD: GET

ENDPOINT: /api/auth/logout/

Response

    {
        "message": "Logged out successfully."
    }
